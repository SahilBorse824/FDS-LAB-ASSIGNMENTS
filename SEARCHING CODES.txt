n=int(input("Enter the total number of students:"))

    rollno.append(int(input("Enter roll no:")))

#USING LINEAR SEARCH
                  
el=int(input("Enter the roll no. of the student to be checked:"))
if(el in rollno):
     print("Roll no.",el,"was present")
else:
     print("Roll no.",el,"was absent")

sentinal_search()



#USING BINARY SEARCH

def binary_search(rollno,el):
    first=0
    last=len(rollno)-1
    found=False
    while first<=last and not found:
        mid=(first+last)/2
        if(rollno[mid]==el):
            found=True
        elif(el<rollno[mid]):
            last=mid-1
        else:
            first=mid+1
    return found
Found=binary_search(rollno,el)
if(Found):
    print("Element found")
else:
    print("Element not found")
